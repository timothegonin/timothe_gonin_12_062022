import PropTypes from 'prop-types'
import styled from 'styled-components'
import {
  RadialBarChart,
  RadialBar,
  ResponsiveContainer,
  PolarAngleAxis,
} from 'recharts'

/**
 * Displaying user's score in RadialBarchart
 * @function Score
 * @param {Array.<{ value: Integer}> } Score User's score data
 * @return {HTMLElement }
 */

/* 
  ┌─────────────────────────────────────────────────────────────────────────┐
  │ JSX                                                                     │
  └─────────────────────────────────────────────────────────────────────────┘
 */
const Score = ({ score }) => {
  return (
    <ScoreWrapper>
      <h3>Score</h3>
      <ScoreChartContent>
        <ScoreChartInfo>
          <p className="score__percent">{score.value}%</p>
          <p>de votre objectif</p>
        </ScoreChartInfo>
        <ResponsiveContainer width="100%" height="100%">
          <RadialBarChart
            width={160}
            height={160}
            innerRadius={90}
            outerRadius={75}
            barSize={10}
            data={[score]}
            transform="rotate(-90 0 0)"
          >
            <PolarAngleAxis
              type="number"
              domain={[0, 100]}
              tick={false}
              fill="red"
            />
            <RadialBar cornerRadius={50} dataKey="value" fill="#FF0000" />
          </RadialBarChart>
        </ResponsiveContainer>
      </ScoreChartContent>
    </ScoreWrapper>
  )
}

export default Score

/* 
  ┌─────────────────────────────────────────────────────────────────────────┐
  │ PROPTYPES                                                               │
  └─────────────────────────────────────────────────────────────────────────┘
 */
Score.propTypes = {
  score: PropTypes.objectOf(PropTypes.number),
}
